def findTotalCost(arr):
    # Sort the array to always access current min and max
    arr.sort()
    l = 0
    r = len(arr) - 1
    total_cost = 0

    while l < r:
        a = arr[l]
        b = arr[r]

        # cost = ceil((a + b) / (b - a + 1))
        denom = b - a + 1
        cost = (a + b + denom - 1) // denom  # integer ceil
        total_cost += cost

        # Replace the max with (a + b)
        arr[r] = a + b

        # Remove the min
        l += 1
        # r stays same (we replaced arr[r] with the new value)

    return total_cost

    arr = []

    for _ in range(arr_count):

    result = findTotalCost(arr)
